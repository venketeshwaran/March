
public class Prime_Number {
	
	public static void main(String[] args) {
		
		int num =3;
		
		int count = 0;
		
		for (int i = 2; i < num; i++) {
			// i=2 2<3
			if (num%i==0) { 
				
				count++; //count =0
			}
		}
		
		if (count>0) { //1>0
			
			System.out.println("not a prime number");
		}
		
		else {
			
			System.out.println("prime number");
			
		}
		
	}

}
